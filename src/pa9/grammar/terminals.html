List of Terminals

OP_COLON: statement,

OP_LPAREN: immediate (x 5),

OP_LSQUARE: memory (x 7),

OP_MINUS: statement, immediate (x 2), memory (x 2),

OP_PLUS: immediate, memory (x 2),

OP_RPAREN: immediate (x 5),

OP_RSQUARE: memory (x 7),

OP_SEMICOLON: program,

TT_IDENTIFIER: label, opcode, register,

TT_LITERAL: statement (x 2), immediate (x 5), memory (x 5),
